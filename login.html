🗄 SQL Schema
-- SkillForge AI Database Schema -- Engine: PostgreSQL / MySQL / SQLite CREATE TABLE users ( id SERIAL PRIMARY KEY, name VARCHAR(120) NOT NULL, email VARCHAR(255) NOT NULL UNIQUE, phone VARCHAR(20), password_hash VARCHAR(255) NOT NULL, created_at TIMESTAMP DEFAULT NOW(), updated_at TIMESTAMP DEFAULT NOW(), is_verified BOOLEAN DEFAULT FALSE, avatar_url TEXT ); CREATE TABLE sessions ( id SERIAL PRIMARY KEY, user_id INT REFERENCES users(id), token VARCHAR(255) NOT NULL UNIQUE, expires_at TIMESTAMP NOT NULL, created_at TIMESTAMP DEFAULT NOW() ); CREATE TABLE submissions ( id SERIAL PRIMARY KEY, user_id INT REFERENCES users(id), task_id VARCHAR(50) NOT NULL, role VARCHAR(50), score INT, link TEXT, feedback TEXT, submitted_at TIMESTAMP DEFAULT NOW() ); -- Indexes for performance CREATE INDEX idx_users_email ON users(email); CREATE INDEX idx_sessions_user ON sessions(user_id); CREATE INDEX idx_subs_user ON submissions(user_id); -- bcrypt hashed passwords only -- Never store plaintext passwords
